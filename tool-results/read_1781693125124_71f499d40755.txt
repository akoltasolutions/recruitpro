   387→  // ─── Copy-Paste helpers ───
   388→  const parsePasteInput = (text: string) => {
   389→    if (!text.trim()) { setPasteParsed([]); setPasteColumns([]); return }
   390→
   391→    const lines = text.trim().split(/\r?\n/)
   392→    const parsed: ManualEntry[] = []
   393→
   394→    for (const line of lines) {
   395→      if (!line.trim()) continue
   396→
   397→      // Strategy 1: Tab-separated (Excel/Sheets copy) — most reliable
   398→      // Strategy 2: Comma-separated (CSV)
   399→      // Strategy 3: Multi-space separated (2+ spaces between fields)
   400→      // Strategy 4: Single-space with phone detection — find first token that
   401→      //   looks like a phone number (6+ digits) and split name/phone there
   402→      let cells: string[]
   403→
   404→      if (line.includes('\t')) {
   405→        cells = line.split('\t').map(c => c.trim())
   406→      } else if (line.includes(',')) {
   407→        cells = line.split(',').map(c => c.trim())
   408→      } else if (/  /.test(line)) {
   409→        // Multi-space: split on 2+ spaces
   410→        cells = line.split(/  +/).map(c => c.trim())
   411→      } else {
   412→        // Single-space: detect phone boundary (first token that is 6+ digits)
   413→        const tokens = line.split(/\s+/)
   414→        const phoneIdx = tokens.findIndex(t => /^\d{6,15}$/.test(t.replace(/[\-+()]/g, '')))
   415→        if (phoneIdx > 0) {
   416→          // Everything before phone = name, phone = phone, rest = role/location/company
   417→          const name = tokens.slice(0, phoneIdx).join(' ')
   418→          const phone = tokens[phoneIdx]
   419→          const rest = tokens.slice(phoneIdx + 1)
   420→          cells = [name, phone, rest[0] || '', rest[1] || '', rest.slice(2).join(' ') || '']
   421→        } else if (phoneIdx === 0) {
   422→          // Phone is first token — use as-is (no name)
   423→          const rest = tokens.slice(1)
   424→          cells = [tokens[0], rest[0] || '', rest[1] || '', rest[2] || '', rest.slice(3).join(' ') || '']
   425→        } else {
   426→          // No phone detected — treat first token as name, second as phone (best effort)
   427→          cells = tokens.length >= 2
   428→            ? [tokens[0], tokens[1], tokens.slice(2).join(' ')]
   429→            : tokens
   430→        }
   431→      }
   432→
   433→      if (cells.length >= 2) {
   434→        parsed.push({
   435→          name: cells[0] || '',
   436→          phone: cells[1] || '',
   437→          role: cells[2] || '',
   438→          location: cells[3] || '',
   439→          company: cells[4] || '',
   440→        })
   441→      }
   442→    }
   443→
   444→    setPasteParsed(parsed)
   445→    setPasteColumns(parsed.length > 0 ? Object.keys(parsed[0]) : [])
   446→  }

   531→    const mapping: Record<string, string> = {}

   533→      const lower = col.toLowerCase()
   534→      if (lower.includes('name') && !lower.includes('client') && !lower.includes('company')) mapping[col] = 'name'
   535→      else if (lower.includes('phone') || lower.includes('mobile') || lower.includes('number')) mapping[col] = 'phone'
   536→      else if (lower.includes('email') || lower.includes('mail')) mapping[col] = 'email'
   537→      else if (lower.includes('role') || lower.includes('position') || lower.includes('title') || lower.includes('job')) mapping[col] = 'role'
   538→      else if (lower.includes('location') || lower.includes('city') || lower.includes('place')) mapping[col] = 'location'
   539→      else if (lower.includes('company') || lower.includes('org') || lower.includes('employer')) mapping[col] = 'company'
   540→    })
