   493→  // ─── CSV / XLSX upload ───
   494→  const autoDetectMapping = (cols: string[]) => {
   495→    const mapping: Record<string, string> = {}
   496→    cols.forEach(col => {
   497→      const lower = col.toLowerCase()
   498→      if (lower.includes('name') && !lower.includes('client') && !lower.includes('company')) mapping[col] = 'name'
   499→      else if (lower.includes('phone') || lower.includes('mobile') || lower.includes('number')) mapping[col] = 'phone'
   500→      else if (lower.includes('email') || lower.includes('mail')) mapping[col] = 'email'
   501→      else if (lower.includes('role') || lower.includes('position') || lower.includes('title') || lower.includes('job')) mapping[col] = 'role'
   502→      else if (lower.includes('location') || lower.includes('city') || lower.includes('place')) mapping[col] = 'location'
   503→      else if (lower.includes('company') || lower.includes('org') || lower.includes('employer')) mapping[col] = 'company'
   504→    })
   505→    return mapping
   506→  }
